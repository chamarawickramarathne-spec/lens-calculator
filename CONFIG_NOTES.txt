// ============================================
// API ENDPOINTS
// ============================================

/*
 * GET /api/get_categories.php
 * - Returns all equipment categories
 * 
 * GET /api/get_equipment.php?category_id=X
 * - Returns equipment for specific category
 * - Omit category_id for all equipment
 * 
 * GET /api/get_templates.php
 * - Returns all package templates with equipment
 * 
 * POST /api/save_package.php
 * - Saves a package to database
 * - Requires JSON body with package details
 * 
 * GET /api/generate_pdf.php?[params]
 * - Generates PDF quotation
 * - Opens in new window for printing
 */
